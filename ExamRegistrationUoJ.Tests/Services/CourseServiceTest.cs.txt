using Xunit;
using ExamRegistrationUoJ.Services;
using System.Collections.Generic;

public class CourseServiceTest
{
    [Fact]
    public void GetAllCourses_ReturnsListOfCourses()
    {
        // Arrange
        var service = new CourseService();

        // Act
        var courses = service.GetAllCourses();

        // Assert
        Assert.Equal(3, courses.Count);
        Assert.Contains("Math", courses);
        Assert.Contains("Science", courses);
        Assert.Contains("History", courses);
    }
}
